设计一个加法满足100位数相加 Java

Feb 7, 2021
1 min read

设计一个加法，满足100位数字的相加

思路：将两个加数以字符串形式保存，倒序每位进行加法，进行进位判断，结束之后再按倒序输出

实现：

package test;

public class Adder {
    public static void BigNumSum(String num1, String num2){ // 定义静态函数
        char[] numA = new StringBuilder(num1).reverse().toString().toCharArray(); //倒序保存num1的数组形式
        char[] numB = new StringBuilder(num2).reverse().toString().toCharArray(); //倒序保存num2的数组形式
        int resultLength = Math.max(numA.length, numB.length); //保存两个数的最大位数
        int[] result = new int[resultLength + 1]; //定义一个位数为 resultLength + 1 的数组，用来保存结果
        for (int i = 0; i < resultLength; ++i) //开始计算
        {
            int Aint = i < numA.length ? (numA[i] - '0') : 0;
            int Bint = i < numB.length ? (numB[i] - '0') : 0; //将A和B转成数字形式，如果i位置没有数字用0代替
            int temp = result[i]; //保存i位置的数字(此位置如果有数字一定是由上一位的加法进位过来的)
            temp += Aint; 
            temp += Bint; //加上A和B
            if(temp >= 10){ //判断进位
                result[i + 1] = temp / 10;
                result[i] = temp % 10;
            }
            else //无需进位则直接保存
                result[i] = temp;
        }
        StringBuilder sb = new StringBuilder(); //用sb输出结果
        if(result[resultLength - 1] != 0){ //如果最高位不为0进行保存，为0则省略
            sb.append(result[resultLength - 1]);
        }
        for(int i = resultLength - 2; i >= 0; --i){ //从倒数第二位开始保存结果
            sb.append(result[i]);
        }
        System.out.println("计算结果为：" + sb); //输出结果
    }

    public static void main(String[] args) {
        BigNumSum("123456789", "6794315"); //验证
    }

}
