自定义连接池

使用原子数组，无锁法(乐观锁)

public class Pool {
    // 连接池大小
    private int poolSize;
    // Connection对象数组
    private Connection[] connections;
    // 连接池状态
    private AtomicIntegerArray status;

    public Pool(int poolSize) {
        this.poolSize = poolSize;
        this.connections = new Connection[poolSize];
        this.status = new AtomicIntegerArray(new int[poolSize]);
        for (int i = 0; i < poolSize; i++) {
            connections[i] = new Connection("conn" + i);
        }
    }

    // 获取连接
    public Connection getConnection() {
        while (true) {
            for (int i = 0; i < poolSize; i++) {
                if (status.get(i) == 0) {
                    if (status.compareAndSet(i, 0, 1)){
                        print("get " + connections[i]);
                        return connections[i];
                    }
                }
            }
            synchronized (this) {
                try {
                    print("wait...");
                    this.wait();
                } catch (InterruptedException e) {
                    e.printStackTrace();
                }
            }
        }
    }

    // 释放连接
    public void free(Connection connection) {
        for (int i = 0; i < poolSize; i++) {
            if (connections[i] == connection) {
                status.set(i, 0);
                synchronized (this) {
                    print("free " + connection);
                    this.notify();
                }
                break;
            }
        }
    }
}

class Connection {
    private String name;

    public Connection(String name) {
        this.name = name;
    }

    @Override
    public String toString() {
        return name;
    }
}

ReentrantLock + Deque 加锁(悲观锁)实现，多条件变量

一、自定义拒绝策略接口

// 拒绝策略，当任务队列满时进行的操作
@FunctionalInterface
interface RejectPolicy<T> {
    void reject(BlockingQueue<T> queue, T task);
}

二、自定义任务队列BlockingQueue

class BlockingQueue<T> {
    // 任务队列
    private Deque<T> queue = new ArrayDeque<>();
    // 锁
    private ReentrantLock lock = new ReentrantLock();
    // 生产者条件变量
    private Condition fullWaitSet = lock.newCondition();
    // 消费者条件变量
    private Condition emptyWaitSet = lock.newCondition();
    // 容量
    private int capacity;


    public BlockingQueue(int capacity) {
        this.capacity = capacity;
    }

    // 阻塞获取任务
    public T take(){
        lock.lock();
        try {
            while (queue.size() == 0){
                try {
                    emptyWaitSet.await();
                } catch (InterruptedException e) {
                    e.printStackTrace();
                }
            }
            T ele = queue.removeFirst();
            fullWaitSet.signal();
            return ele;
        } finally {
            lock.unlock();
        }
    }

    // 阻塞添加任务
    public void put(T task){
        lock.lock();
        try {
            while (queue.size() == capacity){
                try {
                    fullWaitSet.await();
                } catch (InterruptedException e) {
                    e.printStackTrace();
                }
            }
            queue.addLast(task);
            emptyWaitSet.signal();
        } finally {
            lock.unlock();
        }
    }

    // 带超时的阻塞获取任务
    public T poll(long timeout, TimeUnit unit){
        lock.lock();
        try {
            // 将timeout统一转换为纳秒
            long nanos = unit.toNanos(timeout);
            while (queue.size() == 0){
                try {
                    if (nanos <= 0){
                        return null;
                    }
                    // 返回的是剩余时间
                    nanos = emptyWaitSet.awaitNanos(nanos);
                } catch (InterruptedException e) {
                    e.printStackTrace();
                }
            }
            T ele = queue.removeFirst();
            fullWaitSet.signal();
            return ele;
        } finally {
            lock.unlock();
        }
    }

    // 带超时的阻塞添加任务
    public boolean offer(T task, long timeout, TimeUnit unit){
        lock.lock();
        try {
            long nanos = unit.toNanos(timeout);
            while (queue.size() == capacity){
                try {
                    if (nanos <= 0){
                        return false;
                    }
                    nanos = fullWaitSet.awaitNanos(nanos);
                } catch (InterruptedException e) {
                    e.printStackTrace();
                }
            }
            queue.addLast(task);
            emptyWaitSet.signal();
            return true;
        } finally {
            lock.unlock();
        }
    }

    public int size(){
        lock.lock();
        try {
            return queue.size();
        } finally {
            lock.unlock();
        }
    }

    // 自定义拒绝策略的添加任务
    public void tryPut(RejectPolicy<T> operation, T task) {
        lock.lock();
        try {
            if (queue.size() == capacity){
                operation.reject(this, task);
            }else {
                print("加入任务队列");
                queue.addLast(task);
                emptyWaitSet.signal();
            }
        } finally {
            lock.unlock();
        }
    }
}

三、自定义线程池ThreadPool

class ThreadPool {
    // 阻塞任务队列
    private BlockingQueue<Runnable> taskQueue;
    // 线程集合 
    private HashSet<Worker> workers = new HashSet<Worker>();
    // 核心线程数量
    private int coreSize;
    // 超时时间
    private long timeout;
    // 超时时间单位
    private TimeUnit timeUnit;
    // 拒绝策略(函数式接口)
    private RejectPolicy<Runnable> operation;

    public ThreadPool(int coreSize, long timeout, TimeUnit timeUnit, int capacity, RejectPolicy<Runnable> operation) {
        this.coreSize = coreSize;
        this.timeout = timeout;
        this.timeUnit = timeUnit;
        this.taskQueue = new BlockingQueue<>(capacity);
        this.operation = operation;
    }

    public void execute(Runnable task){
        /*
        当任务数量没有超过 coreSize 时，直接交给worker执行
        当任务数量超过 coreSize 时，加入任务队列
         */
        synchronized (workers) {
            if(workers.size() < coreSize){
                Worker worker = new Worker(task);
                print("创建新worker：" + worker);
                workers.add(worker);
                worker.start();
            }else {
                taskQueue.tryPut(operation, task);
            }
        }
    }

    // worker内部类，对线程进行封装
    class Worker extends Thread{
        private Runnable task;
        
        public Worker(Runnable task) {
            this.task = task;
        }

        @Override
        public void run() {
            // 执行任务
            // 如果task不为空，直接执行
            while (null != task || (task = taskQueue.poll(timeout, timeUnit)) != null){
                try {
                    print(this + " worker开始执行任务");
                    task.run();
                } catch (Exception e){
                    e.printStackTrace();
                } finally {
                    task = null;
                }
            }
            // 如果task空，从任务队列获取任务执行
            synchronized (workers){
                print(this + "worker完成任务");
                workers.remove(this);
            }
        }
    }
}
